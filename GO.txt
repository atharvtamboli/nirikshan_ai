package main

import (
	"crypto/aes"
	"crypto/cipher"
	"crypto/rand"
	"crypto/sha256"
	"encoding/hex"
	"fmt"
	"io"
	"io/ioutil"
	"time"

	"github.com/gin-gonic/gin"
)

var encryptionKey = []byte("your-32-byte-super-secure-key!")

// Block structure
type Block struct {
	Index     int    `json:"index"`
	Timestamp string `json:"timestamp"`
	FileName  string `json:"file_name"`
	FileHash  string `json:"file_hash"`
	PrevHash  string `json:"prev_hash"`
	Hash      string `json:"hash"`
}

// Blockchain storage
var Blockchain []Block

// Encrypt file content
func encryptFile(data []byte) ([]byte, error) {
	block, err := aes.NewCipher(encryptionKey)
	if err != nil {
		return nil, err
	}

	ciphertext := make([]byte, aes.BlockSize+len(data))
	iv := ciphertext[:aes.BlockSize]
	if _, err := io.ReadFull(rand.Reader, iv); err != nil {
		return nil, err
	}

	stream := cipher.NewCFBEncrypter(block, iv)
	stream.XORKeyStream(ciphertext[aes.BlockSize:], data)

	return ciphertext, nil
}

// Calculate hash of the block
func calculateHash(block Block) string {
	record := fmt.Sprintf("%d%s%s%s%s", block.Index, block.Timestamp, block.FileName, block.FileHash, block.PrevHash)
	hash := sha256.Sum256([]byte(record))
	return hex.EncodeToString(hash[:])
}

// Generate a hash of a file
func getFileHash(filePath string) (string, error) {
	fileData, err := ioutil.ReadFile(filePath)
	if err != nil {
		return "", err
	}
	hash := sha256.Sum256(fileData)
	return hex.EncodeToString(hash[:]), nil
}

// Create a new block
func createBlock(prevBlock Block, fileName string, fileHash string) Block {
	newBlock := Block{
		Index:     prevBlock.Index + 1,
		Timestamp: time.Now().String(),
		FileName:  fileName,
		FileHash:  fileHash,
		PrevHash:  prevBlock.Hash,
	}
	newBlock.Hash = calculateHash(newBlock)
	return newBlock
}

// Create the Genesis block
func createGenesisBlock() Block {
	return Block{Index: 0, Timestamp: time.Now().String(), FileName: "Genesis Block", FileHash: "0", PrevHash: "0", Hash: ""}
}

func main() {
	// Initialize blockchain with the Genesis block
	Blockchain = append(Blockchain, createGenesisBlock())

	r := gin.Default()
	r.Static("/static", "./static")
	r.GET("/", func(c *gin.Context) {
		c.File("index.html")
	})

	// Upload a file and encrypt before storing on blockchain
	r.POST("/upload", func(c *gin.Context) {
		file, err := c.FormFile("file")
		if err != nil {
			fmt.Printf("File upload error: %v\n", err)
			c.JSON(400, gin.H{"error": "No file uploaded"})
			return
		}
		fmt.Printf("File received: %s\n", file.Filename)

		filePath := "uploads/" + file.Filename
		if err := c.SaveUploadedFile(file, filePath); err != nil {
			fmt.Printf("Save file error: %v\n", err)
			c.JSON(500, gin.H{"error": "Failed to save file"})
			return
		}
		fmt.Printf("File saved to: %s\n", filePath)

		fileData, err := ioutil.ReadFile(filePath)
		if err != nil {
			fmt.Printf("Read file error: %v\n", err)
			c.JSON(500, gin.H{"error": "Failed to read file"})
			return
		}
		fmt.Printf("File read successfully, size: %d bytes\n", len(fileData))

		encryptedData, err := encryptFile(fileData)
		if err != nil {
			fmt.Printf("Encryption error: %v\n", err)
			c.JSON(500, gin.H{"error": "Encryption failed"})
			return
		}
		fmt.Printf("File encrypted successfully\n")

		if err := ioutil.WriteFile(filePath, encryptedData, 0644); err != nil {
			fmt.Printf("Write encrypted file error: %v\n", err)
			c.JSON(500, gin.H{"error": "Failed to save encrypted file"})
			return
		}

		fileHash, err := getFileHash(filePath)
		if err != nil {
			fmt.Printf("Hash calculation error: %v\n", err)
			c.JSON(500, gin.H{"error": "Failed to hash file"})
			return
		}
		fmt.Printf("File hash calculated: %s\n", fileHash)

		fmt.Printf("Current blockchain length: %d\n", len(Blockchain))
		newBlock := createBlock(Blockchain[len(Blockchain)-1], file.Filename, fileHash)
		Blockchain = append(Blockchain, newBlock)
		fmt.Printf("New blockchain length: %d\n", len(Blockchain))
		fmt.Printf("New block added: %+v\n", newBlock)

		c.JSON(200, gin.H{"message": "File encrypted and stored on blockchain", "file_name": file.Filename, "hash": newBlock.Hash})
	})

	// Get blockchain data
	r.GET("/blocks", func(c *gin.Context) {
		c.JSON(200, Blockchain)
	})

	r.Run(":8080")
}
